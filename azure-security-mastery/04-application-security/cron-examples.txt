# ⏰ Cron Job Examples for Container Scheduling
# Add these to your crontab with: crontab -e

# Start container every day at 9 AM
0 9 * * * cd /home/yom/cybersecurity-journey/azure-security-mastery/04-application-security && ./start-minishell.sh >> /tmp/minishell-start.log 2>&1

# Stop container every day at 6 PM (after 9 hours)
0 18 * * * cd /home/yom/cybersecurity-journey/azure-security-mastery/04-application-security && ./stop-minishell.sh >> /tmp/minishell-stop.log 2>&1

# Alternative: Weekend-only schedule
# Start Saturday 10 AM, stop Sunday 8 PM
0 10 * * 6 cd /home/yom/cybersecurity-journey/azure-security-mastery/04-application-security && ./start-minishell.sh >> /tmp/minishell-start.log 2>&1
0 20 * * 0 cd /home/yom/cybersecurity-journey/azure-security-mastery/04-application-security && ./stop-minishell.sh >> /tmp/minishell-stop.log 2>&1

# 3. Save and exit
